# Androidの使用例

## MainActivityの実装

```kotlin
// アプリ全体で単一インスタンスを共有
object SdkConfig {
    val sdk = KlleonOndeviceSdk()
}

class MainActivity : ComponentActivity() {
    private val klleonOndeviceSdk = SdkConfig.sdk
    private val resourceManager = ResourceManager()

    private val requiredPermissions = arrayOf(
        Manifest.permission.RECORD_AUDIO
    )

    private val permissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestMultiplePermissions()
    ) { permissions ->
        handlePermissionResults(permissions)
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        if (checkPermissions()) {
            initializeApp()
        } else {
            requestPermissions()
        }
    }

    private fun initializeApp() {
        setContent {
            OndeviceSampleTheme {
                Surface(
                    modifier = Modifier.fillMaxSize(),
                    color = MaterialTheme.colorScheme.background
                ) {
                    WorkerScreen(klleonOndeviceSdk = klleonOndeviceSdk)
                }
            }
        }

        // リソース初期化（キャッシュコピー + YUV変換 + メモリロード）
        resourceManager.init(
            context = this,
            mp4FileName = "loop_sample.mp4",
            npzFileName = "loop_sample.npz",
            styleFileName = "loop_sample.style",
            avatarId = "character_001",
            onSuccess = {
                // リソース準備完了 — play()の呼び出しが可能です
            },
            onError = { error ->
                Log.e("MainActivity", "リソース準備失敗", error)
            }
        )
    }
}
```

## Compose UI統合

### WorkerScreen Composable

```kotlin
@Composable
fun WorkerScreen(
    klleonOndeviceSdk: KlleonOndeviceSdk,
    viewModel: WorkerViewModel = viewModel()
) {
    val context = LocalContext.current
    var workerView by remember { mutableStateOf<KlleonSdkView?>(null) }
    var messageData by remember { mutableStateOf("") }

    // SDK状態の収集
    val accumulatedMessages by viewModel.accumulatedMessages.collectAsState()

    LaunchedEffect(Unit) {
        viewModel.observeSdkState(klleonOndeviceSdk)
    }

    Box(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp)
    ) {
        // SDK状態の表示
        Card(
            modifier = Modifier
                .align(Alignment.TopStart)
                .padding(8.dp),
            colors = CardDefaults.cardColors(
                containerColor = Color.Transparent
            )
        ) {
            Column(modifier = Modifier.padding(12.dp)) {
                Text(
                    text = accumulatedMessages,
                    style = MaterialTheme.typography.bodySmall,
                    color = Color.Black.copy(alpha = 0.8f)
                )
            }
        }

        // 下部コンテンツ
        Column(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .fillMaxWidth(),
            verticalArrangement = Arrangement.spacedBy(16.dp)
        ) {
            // SDKビュー
            Card(
                modifier = Modifier
                    .fillMaxWidth()
                    .height(400.dp)
            ) {
                AndroidView(
                    factory = { ctx ->
                        KlleonSdkView(ctx).also { view ->
                            workerView = view
                        }
                    },
                    modifier = Modifier.fillMaxSize()
                )
            }

            // メッセージ入力
            OutlinedTextField(
                value = messageData,
                onValueChange = { messageData = it },
                label = { Text("メッセージ") },
                modifier = Modifier.fillMaxWidth()
            )

            // 制御ボタン
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        workerView?.let { view ->
                            CoroutineScope(Dispatchers.IO).launch {
                                klleonOndeviceSdk.play(
                                    context = context,
                                    klleonSdkViewInstance = view,
                                    sdkKey = "YOUR_SDK_KEY",
                                    avatarId = "character_001",
                                    language = "ko_kr"
                                )
                            }
                        }
                    },
                    modifier = Modifier.weight(1f),
                    enabled = workerView != null
                ) {
                    Text("再生")
                }

                Button(
                    onClick = {
                        CoroutineScope(Dispatchers.Default).launch {
                            klleonOndeviceSdk.finish()
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("終了")
                }
            }

            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        klleonOndeviceSdk.sendMessage(messageData)
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("メッセージ送信")
                }

                Button(
                    onClick = {
                        klleonOndeviceSdk.sendMessageEcho(messageData)
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("エコー")
                }

                Button(
                    onClick = {
                        klleonOndeviceSdk.sendStopSpeech()
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("発話中断")
                }
            }

            // 音声認識ボタン
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        if (klleonOndeviceSdk.isVoiceRecognizing()) {
                            klleonOndeviceSdk.stopVoiceRecognition()
                        } else {
                            klleonOndeviceSdk.startVoiceRecognition()
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("音声認識")
                }

                Button(
                    onClick = {
                        klleonOndeviceSdk.sendChangeLanguage("en")
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("英語に切替")
                }
            }
        }
    }
}
```

## ViewModelの実装

### WorkerViewModel

```kotlin
class WorkerViewModel : ViewModel() {

    private val _accumulatedMessages = MutableStateFlow("")
    val accumulatedMessages: StateFlow<String> = _accumulatedMessages.asStateFlow()

    fun observeSdkState(klleonOndeviceSdk: KlleonOndeviceSdk) {
        viewModelScope.launch {
            klleonOndeviceSdk.sdkState.collect { state ->
                val messageToAdd = try {
                    val json = JSONObject(state)
                    val chatType = json.optString("chat_type", "")
                    val message = json.getString("message")

                    when (chatType) {
                        "STT_RESULT" -> "[STT] $message"
                        "STT_PARTIAL" -> "[STT中間] $message"
                        "RESPONSE" -> "[アバター] $message"
                        else -> message
                    }
                } catch (t: Throwable) {
                    state.toString()
                }

                if (messageToAdd.isNotBlank()) {
                    val currentMessages = _accumulatedMessages.value
                    _accumulatedMessages.value = if (currentMessages.isEmpty()) {
                        messageToAdd
                    } else {
                        "$currentMessages\n$messageToAdd"
                    }
                }
            }
        }
    }

    fun clearMessages() {
        _accumulatedMessages.value = ""
    }
}
```

## アバター切り替えの例

`init()`はアバターの変更を自動検出します。同じアバターであればバッファを即座に再利用し、異なるアバターであれば以前のバッファを解放後に新たに割り当てます。

```kotlin
class CharacterSelectActivity : ComponentActivity() {

    fun onAvatarSelected(avatarMp4: String, avatarNpz: String, avatarId: String) {
        // init()がavatarIdを比較して自動でバッファの再利用/切り替えを処理
        // 同じアバター: 即座にonSuccess（0ms）
        // 異なるアバター: 以前のバッファ解放 + 新規割り当て
        ResourceManager.init(
            context = this,
            mp4FileName = avatarMp4,
            npzFileName = avatarNpz,
            avatarId = avatarId,
            onSuccess = {
                // リソース準備完了 — ChatActivityへ遷移
                startActivity(Intent(this, ChatActivity::class.java))
            },
            onError = { error ->
                Log.e("CharacterSelect", "リソース準備失敗", error)
            }
        )
    }

    override fun onDestroy() {
        super.onDestroy()
        // play()前に離脱した場合のメモリ解放
        if (!KlleonOndeviceSdk.playing) {
            ResourceManager.deinit()
        }
    }
}
```

## 状態監視パターン

```kotlin
@Composable
fun SDKStateObserver(klleonOndeviceSdk: KlleonOndeviceSdk) {
    var currentState by remember { mutableStateOf("") }

    LaunchedEffect(klleonOndeviceSdk) {
        klleonOndeviceSdk.sdkState.collect { state ->
            currentState = try {
                val json = JSONObject(state)
                "状態: ${json.optString("chat_type")} - ${json.getString("message")}"
            } catch (e: Exception) {
                "生の状態: $state"
            }
        }
    }

    Text(
        text = currentState,
        modifier = Modifier.padding(16.dp)
    )
}
```
