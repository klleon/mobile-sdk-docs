# Android Usage Examples

## MainActivity Implementation

```kotlin
// Share a single instance across the entire app
object SdkConfig {
    val sdk = KlleonOndeviceSdk()
}

class MainActivity : ComponentActivity() {
    private val klleonOndeviceSdk = SdkConfig.sdk
    private val resourceManager = ResourceManager()

    private val requiredPermissions = arrayOf(
        Manifest.permission.RECORD_AUDIO
    )

    private val permissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestMultiplePermissions()
    ) { permissions ->
        handlePermissionResults(permissions)
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        if (checkPermissions()) {
            initializeApp()
        } else {
            requestPermissions()
        }
    }

    private fun initializeApp() {
        setContent {
            OndeviceSampleTheme {
                Surface(
                    modifier = Modifier.fillMaxSize(),
                    color = MaterialTheme.colorScheme.background
                ) {
                    WorkerScreen(klleonOndeviceSdk = klleonOndeviceSdk)
                }
            }
        }

        // Initialize resources (cache copy + YUV conversion + memory loading)
        resourceManager.init(
            context = this,
            mp4FileName = "loop_sample.mp4",
            npzFileName = "loop_sample.npz",
            styleFileName = "loop_sample.style",
            avatarId = "character_001",
            onSuccess = {
                // Resources ready — play() can be called
            },
            onError = { error ->
                Log.e("MainActivity", "Resource preparation failed", error)
            }
        )
    }
}
```

## Compose UI Integration

### WorkerScreen Composable

```kotlin
@Composable
fun WorkerScreen(
    klleonOndeviceSdk: KlleonOndeviceSdk,
    viewModel: WorkerViewModel = viewModel()
) {
    val context = LocalContext.current
    var workerView by remember { mutableStateOf<KlleonSdkView?>(null) }
    var messageData by remember { mutableStateOf("") }

    // Collect SDK state
    val accumulatedMessages by viewModel.accumulatedMessages.collectAsState()

    LaunchedEffect(Unit) {
        viewModel.observeSdkState(klleonOndeviceSdk)
    }

    Box(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp)
    ) {
        // SDK state display
        Card(
            modifier = Modifier
                .align(Alignment.TopStart)
                .padding(8.dp),
            colors = CardDefaults.cardColors(
                containerColor = Color.Transparent
            )
        ) {
            Column(modifier = Modifier.padding(12.dp)) {
                Text(
                    text = accumulatedMessages,
                    style = MaterialTheme.typography.bodySmall,
                    color = Color.Black.copy(alpha = 0.8f)
                )
            }
        }

        // Bottom content
        Column(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .fillMaxWidth(),
            verticalArrangement = Arrangement.spacedBy(16.dp)
        ) {
            // SDK view
            Card(
                modifier = Modifier
                    .fillMaxWidth()
                    .height(400.dp)
            ) {
                AndroidView(
                    factory = { ctx ->
                        KlleonSdkView(ctx).also { view ->
                            workerView = view
                        }
                    },
                    modifier = Modifier.fillMaxSize()
                )
            }

            // Message input
            OutlinedTextField(
                value = messageData,
                onValueChange = { messageData = it },
                label = { Text("Message") },
                modifier = Modifier.fillMaxWidth()
            )

            // Control buttons
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        workerView?.let { view ->
                            CoroutineScope(Dispatchers.IO).launch {
                                klleonOndeviceSdk.play(
                                    context = context,
                                    klleonSdkViewInstance = view,
                                    sdkKey = "YOUR_SDK_KEY",
                                    avatarId = "character_001",
                                    language = "ko_kr"
                                )
                                // Auto standby after 60 seconds of inactivity
                                klleonOndeviceSdk.setIdleTimeout()
                            }
                        }
                    },
                    modifier = Modifier.weight(1f),
                    enabled = workerView != null
                ) {
                    Text("Play")
                }

                Button(
                    onClick = {
                        CoroutineScope(Dispatchers.Default).launch {
                            klleonOndeviceSdk.finish()
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Stop")
                }
            }

            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        klleonOndeviceSdk.sendMessage(messageData)
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Send Message")
                }

                Button(
                    onClick = {
                        klleonOndeviceSdk.sendMessageEcho(messageData)
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Echo")
                }

                Button(
                    onClick = {
                        klleonOndeviceSdk.sendStopSpeech()
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Stop Speech")
                }
            }

            // Voice recognition buttons
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        if (klleonOndeviceSdk.isVoiceRecognizing()) {
                            klleonOndeviceSdk.stopVoiceRecognition()
                        } else {
                            klleonOndeviceSdk.startVoiceRecognition()
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Voice Recognition")
                }

                Button(
                    onClick = {
                        klleonOndeviceSdk.sendChangeLanguage("en")
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("Switch to English")
                }
            }
        }
    }
}
```

## Standby Mode

### Exiting Standby with Wake-word

An example of exiting standby via voice commands combined with wakeword matching.
Wakeword matching is automatically enabled upon `enterStandby()` and automatically disabled upon `exitStandby()`.

```kotlin
// Register wakewords — registration alone does not activate matching (auto-activated on standby entry)
klleonOndeviceSdk.registerWakewords(mapOf(
    "ジャパニーズ" to "ja_jp",
    "イングリッシュ" to "en_us",
    "コリアン" to "ko_kr",
))

// Switch language + exit standby when wakeword is detected
LaunchedEffect(Unit) {
    klleonOndeviceSdk.wakewordDetected.collect { result ->
        klleonOndeviceSdk.sendChangeLanguage(result.tag)
        if (klleonOndeviceSdk.isStandby()) {
            klleonOndeviceSdk.exitStandby()  // Matching is also automatically disabled
        }
    }
}
```

## ViewModel Implementation

### WorkerViewModel

```kotlin
class WorkerViewModel : ViewModel() {

    private val _accumulatedMessages = MutableStateFlow("")
    val accumulatedMessages: StateFlow<String> = _accumulatedMessages.asStateFlow()

    fun observeSdkState(klleonOndeviceSdk: KlleonOndeviceSdk) {
        viewModelScope.launch {
            klleonOndeviceSdk.sdkState.collect { state ->
                val messageToAdd = try {
                    val json = JSONObject(state)
                    val chatType = json.optString("chat_type", "")
                    val message = json.getString("message")

                    when (chatType) {
                        "STT_RESULT" -> "[STT] $message"
                        "STT_PARTIAL" -> "[STT Interim] $message"
                        "RESPONSE" -> "[Avatar] $message"
                        else -> message
                    }
                } catch (t: Throwable) {
                    state.toString()
                }

                if (messageToAdd.isNotBlank()) {
                    val currentMessages = _accumulatedMessages.value
                    _accumulatedMessages.value = if (currentMessages.isEmpty()) {
                        messageToAdd
                    } else {
                        "$currentMessages\n$messageToAdd"
                    }
                }
            }
        }
    }

    fun clearMessages() {
        _accumulatedMessages.value = ""
    }
}
```

## Avatar Switching Example

`init()` automatically detects avatar changes. For the same avatar, buffers are reused immediately; for a different avatar, the previous buffer is released before allocating a new one.

```kotlin
class CharacterSelectActivity : ComponentActivity() {

    fun onAvatarSelected(avatarMp4: String, avatarNpz: String, avatarId: String) {
        // init() compares avatarId and automatically handles buffer reuse/switching
        // Same avatar: immediate onSuccess (0ms)
        // Different avatar: release previous buffer + new allocation
        ResourceManager.init(
            context = this,
            mp4FileName = avatarMp4,
            npzFileName = avatarNpz,
            avatarId = avatarId,
            onSuccess = {
                // Resources ready — navigate to ChatActivity
                startActivity(Intent(this, ChatActivity::class.java))
            },
            onError = { error ->
                Log.e("CharacterSelect", "Resource preparation failed", error)
            }
        )
    }

    override fun onDestroy() {
        super.onDestroy()
        // Release memory if exiting before play()
        if (!KlleonOndeviceSdk.playing) {
            ResourceManager.deinit()
        }
    }
}
```

```kotlin
fun SDKStateObserver(klleonOndeviceSdk: KlleonOndeviceSdk) {

    )
```

## Wakeword-based Language Switching (Standby Integration)

An example of automatically switching the avatar's conversation language and exiting standby by detecting voice wakewords in standby mode.

:::info STT Language and Wakeword Registration Strategy
The language passed to `play()` or `sendChangeLanguage()` is also set as the STT recognition language. Wakewords must be registered as **words that the current STT language can recognize**.

For example, in Korean mode (`ko_kr`), Korean pronunciation wakewords like `"니혼고"`, `"잉글리쉬"` are recognized, and when switched to Japanese mode (`ja_jp`), Japanese wakewords like `"韓国語"`, `"英語"` are recognized.

Therefore, it is recommended to **register wakewords for all supported languages at once**.
:::

:::caution Standby Mode Required
Wakeword matching is disabled by default and is automatically activated upon `enterStandby()` entry. You must set up automatic standby with `setIdleTimeout()` or manually call `enterStandby()` for wakeword matching to work.
:::

```kotlin
@Composable
fun WakewordLanguageSwitcher(klleonOndeviceSdk: KlleonOndeviceSdk) {
    var currentLanguage by remember { mutableStateOf("ko_kr") }

    // Register wakewords recognizable in all languages at once
    LaunchedEffect(Unit) {
        klleonOndeviceSdk.registerWakewords(mapOf(
            // Recognized by Korean STT
            "니혼고" to "ja_jp", "일본어" to "ja_jp",
            "잉글리쉬" to "en_us", "영어" to "en_us",
            // Recognized by Japanese STT
            "韓国語" to "ko_kr", "コリアン" to "ko_kr",
            "英語" to "en_us",
            // Recognized by English STT
            "Korean" to "ko_kr",
            "Japanese" to "ja_jp",
        ))

        // Enable automatic standby — wakeword matching is also auto-activated on standby entry
        klleonOndeviceSdk.setIdleTimeout(60_000L)
    }

    // Switch language + exit standby when wakeword is detected
    LaunchedEffect(Unit) {
        klleonOndeviceSdk.wakewordDetected.collect { result ->
            klleonOndeviceSdk.sendChangeLanguage(result.tag)
            currentLanguage = result.tag
            if (klleonOndeviceSdk.isStandby()) {
                klleonOndeviceSdk.exitStandby()  // Matching is also automatically disabled
            }
        }
    }

    Text(
        text = "Current language: $currentLanguage",
        modifier = Modifier.padding(16.dp)
    )
}
```
